int n = number of bins
Dustbin dbins[n]

int current_location

#define CONSTANT    1   // A Constant to control the afinity to give preference to distance and level to which dustbin is full

Dustbin getNextBin()
{
    int a = 0, b = 0;
    Dustbin* tbin;
    for i = 0; i < n; i++ :
    {
        a = abs(dbins[i].distance - current_location);  // abs function returns absolute
        a = CONSTANT * (dbins[i].fill_level / a);
        if(a > b)
        {
            b = a;
            tbin = &dbins[i];
        }
    }
    return tbin;
}

disposeBin(int d)
{
    current_location = d;
}

void main()
{
    Dustbin bin;
    while(1)
    {
        bin = getNextBin();
        disposeBin(bin);
    }
}
